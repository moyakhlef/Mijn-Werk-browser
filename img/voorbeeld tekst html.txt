<script>
    function goTo(screenId) {
        document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
        const target = document.getElementById(screenId);
        if(target) { target.classList.add('active'); window.scrollTo(0, 0); }
    }

    function goBack() {
        const current = document.querySelector('.screen.active');
        if (!current) return;
        const id = current.id;
        
        if (id === 'scr-begindatum') goTo('scr-vrijvragen');
        else if (['scr-overdezeapp', 'scr-voorkeuren', 'scr-profieldetails'].includes(id)) goTo('scr-profiel');
        else if (id === 'scr-profiel') goTo('scr-vandaag');
        else goTo('scr-service');
    }
</script>
